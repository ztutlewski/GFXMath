863  SSE_MAT_CALL PerspectiveProjectionMatrix(float near, float far, float fov, float aspect)
864  {
865  SseMat44 result;
866  static const SseVec NEGATIVE_UNIT_0001 = Vec4Negate(UNIT_0001);
867

878  tmp0 = VecSwizzle<VecCoord::X, Y, W, Z>(tmp0);
