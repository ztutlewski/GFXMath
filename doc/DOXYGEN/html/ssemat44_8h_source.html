853  SSE_MAT_CALL PerspectiveProjectionMatrix(float near, float far, float fov, float aspect)
854  {
855  SseMat44 result;
856  static const SseVec NEGATIVE_UNIT_0001 = Vec4Negate(UNIT_0001);
857

868  tmp0 = VecSwizzle<VecCoord::X, Y, W, Z>(tmp0);
869  result.col3 = VecBlend<LEFT, LEFT, RIGHT, LEFT>(zero, tmp0);// [0.0f 0.0f -fn/(f-n) 0.0f]
